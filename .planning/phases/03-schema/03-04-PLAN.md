```bash
python3 << 'EOF'
import json, re

files = [
    '/Users/claude/Projekte/BesserSehenLandshut/public/index.html',
    '/Users/claude/Projekte/BesserSehenLandshut/public/leistungen-kontaktlinsen.html',
    '/Users/claude/Projekte/BesserSehenLandshut/public/leistungen-nachtlinsen.html',
    '/Users/claude/Projekte/BesserSehenLandshut/public/leistungen-beratung-kurzsichtigkeit.html',
    '/Users/claude/Projekte/BesserSehenLandshut/public/leistungen-visuelle-reha.html',
    '/Users/claude/Projekte/BesserSehenLandshut/public/team.html',
]

errors = []
for path in files:
    fname = path.split('/')[-1]
    with open(path) as f:
        content = f.read()
    # Alle <script type="application/ld+json">...</script> Bloecke extrahieren
    blocks = re.findall(r'<script[^>]+application/ld\+json[^>]*>(.*?)</script>', content, re.DOTALL)
    for i, block in enumerate(blocks):
        block = block.strip()
        try:
            data = json.loads(block)
            if isinstance(data, list):
                types = [obj.get('@type','?') for obj in data]
                print(f'OK: {fname} Block {i+1}: Array mit {len(data)} Eintraegen ({types})')
            else:
                print(f'OK: {fname} Block {i+1}: @type={data.get(\"@type\",\"?\")}')
        except json.JSONDecodeError as e:
            errors.append(f'FEHLER: {fname} Block {i+1}: {e}')
            print(f'FEHLER: {fname} Block {i+1}: {e}')

if errors:
    print(f'\n{len(errors)} Fehler gefunden!')
    raise SystemExit(1)
else:
    print(f'\nAlle JSON-LD-Bloecke sind valides JSON.')
EOF
```

Falls JSON-Fehler auftreten: Syntaxfehler in der betroffenen Datei beheben (fehlende Kommas, unkorrekte Anfuehrungszeichen, etc.) und erneut committen + pushen.
